#include <iostream>
using namespace std;

int main() {
   int numInts;
   cin >> numInts;
   int userInts[numInts];   // An array to hold the user's input integers
   int i;
   
   for (i=0; i<numInts; ++i) {
      cin >> userInts[i];
   }
   
   for (i=numInts-1; i>=0; --i) {
      cout << userInts[i] << " ";
   }
   cout << endl;
   return 0;
}
